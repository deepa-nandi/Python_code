import random as r

num = r.randrange(1,10)
guess = int(input("Enter a number: "))

while (num!=guess):
    if (guess<num):
        print ("Number is too low.")
        guess = int(input("Enter the number again: "))
    elif (guess>num):
        print ("Number is too high.")
        guess = int(input("Enter the number again: "))
    else:
        break
print ("Right guess. The guessed number is: ", num)
